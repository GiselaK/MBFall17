                    //Basic Function: 
                    // Create function 
                    function myFunc () { 
                        return "called"; 
                    }
                    // Call function 
                    myFunc(); -> "called" 

                // With Parameters:
                    // Create function 
                    function myFunc (str) { 
                        return str; 
                    }
                    // Call function 
                    myFunc("called"); -> "called" 
                     // Call function 
                    myFunc("abc"); -> "abc"
  • For Loops More Review: 
     
                    function loop (){
                        // Start at 0, stop at 5, add 1 each time
                        for(var i = 0; i < 5; i++) { 
                            console.log(i); 
                        }
                    }

                    loop(); // 0, 1, 2, 3, 4
  • Arrays More Review 
                        // Create Array: 
                    var array = ["a","b","c"]; 
                    // Access Array: 
                    array[0]; -> a 
                    array[1]; -> b 
                    // Iterate(loop) over an array: 
                    function loop () { 
                        // Start at 0, stop at size of array, add 1 each time
                        for(var i = 0; i < array.length; i++) { 
                            console.log(array[i]); 
                        } 
                    } 
                    loop() -> a b c
